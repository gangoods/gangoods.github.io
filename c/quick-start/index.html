    • 导入标准的输入输出库

    • /* 块注释 */

    • int 表示函数的返回值类型
    • 函数的参数类型 void类型 :表示空类型,void 可以省略不写

    char name[64];
    int age;
    int id;
    double s;

    // 每次打印一下,就要拷贝结构体 差不多80个字节给形参d,
    // 太浪费了
    printf("%s\n",d.name);

void printdog2(struct dog * d){
    // 这个只要传递8个字节的指针就可以了

void printdog3(const struct dog * d){
    // 这个就好了
    printf("%s\n",d->name);
};
// c语言中 一般传递地址,就是指针,,如果要禁止修改内容 则加上const

    struct dog d={"lllllaaaa",10,100,99};
    //
    printdog(d);
    return 0;

}

void reverse_string(char* str){

    if (str == NULL){
        return;
    }

    int begin = 0;
    int end = strlen(str) - 1;

    while (begin < end){

        //交换两个字符元素
        char temp = str[begin];
        str[begin] = str[end];
        str[end] = temp;
        begin++;
        end--;
    }

}
