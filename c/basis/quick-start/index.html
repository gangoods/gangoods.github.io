    char name[64];

    int id;

void print3(const struct dog * d){
    // 这个就好了
    printf("%s\n",d->name);
};

    #include <stdio.h>
#include <string.h>
//字符串拷贝 方法一
void strcpy1(char *dst,char *src){
    int len=strlen(src); // strlen 是不包含\0的
    for (int i=0;i<len;++i){
        dst[i]=src[i];
    }
    dst[len]='\0';
}
// 方法二
void strcpy2(char *dst,char *src)
{
    while (*src!='\0'){
        *dst=*src;
        ++src;
        ++dst;
    }
    *dst="\0";
}

//方法三
void strcpy3(char *dst,char *src)
{
    //等到dst= src的最后一个 就是\0 时,while条件就不符合了,结束了
    while (*dst++=*src++);
}
int main(int argc, char const *argv[])
{
    char src[]="hello";
    char dst[10];
    if (strcpy(dst,src)!=NULL){
        printf("strcpy: %s\n",dst);
    }

    // 注意有限拷贝,不会将\0拷贝到目标字符数组中
    if (strncpy(dst,src,3)!=NULL){
        dst[3]= '\0'; //需要加上这个
        printf("strncpy %s\n",dst);
    }
}

    #include <stdio.h>
#include <string.h>
int main(int argc, char const *argv[])
{
    char s[]="hello";
    const char c='l';//传递字符的ascii 码 数字

    printf("%p\n",p1); // 和arr 的一样
    printf("%s\n",p1); //www
    // 会修改原始字符串数组
    printf("%s\n",arr); //www
    // 可以说明 arr字符串第一个. 被修改为\0,后面的没有变化
    printf("left of arr:%s\n",&arr[4]); //baidu.com

    // 第一个指针a用来指向函数返回的字符串
    // 第二个指针b用来指向原字符串被分割的位置的那个地址. 这里例子就是baidu.com的b位置

    while (p1!=NULL){
        printf("%s\n",p1); // 先打印www, 然后baidu ,最后com
        p1=strtok(NULL,".");
    }
    return 0;
}

    #include <stdio.h>
#include <string.h>
#include <stdlib.h>
// 字符串反转
void strreverse(char* str){
    if (str == NULL){
        return;
    }
    int begin = 0;
    int end = strlen(str) - 1;
    while (begin < end){
        //交换两个字符元素
        char temp = str[begin];
        str[begin] = str[end];
        str[end] = temp;
        begin++;
        end--;
    }
}
