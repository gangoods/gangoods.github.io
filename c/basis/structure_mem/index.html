#include <string.h>
#include <stdlib.h>

void show(int a,int b ,int c){

    // 函数参数创建的变量是在栈中创建的
    // 而栈是从上往下的? 不同系统不一样??
    printf("%p\n",&a);
    printf("%p\n",&b);
    printf("%p\n",&c);
}
// 全局变量,作用范围在整个项目中,
int a=100;
// 如果你的这个全局变量在 其他文件里
// 则这里需要申明这个变量,后面的才能用到
// 全局变量写到main函数下面 也需要申明
// extern int a;

// 静态全局变量 int 未赋值 则默认初始化为0
// 静态 如果放到其他文件中,这里想要使用,是不能的
// 他只能在他所在的文件里使用, 申明带static会报错
static int b=200;
// 函数好像都是全局函数, 如果加上static 变成静态函数
// 那么他的作用域只在所在的文件里
// static extern static f  这个会报错的

// 全局函数
void f(){
    // 静态局部变量
    static int aa=1;
    aa++;
    //函数结束后,没有释放该静态变量,会继续
    printf("%d\n",aa);
}
int main(int argc, char const *argv[])
{
    printf("%d\n",a);
    printf("%d\n",b);
    // 局部变量的定义
    auto int bb=11;// 一般我们省略
    //作用范围:从创建到所在函数结束
    int a=10;
    int b=20;
    int c=30;
    printf("%p\n",&a);
    printf("%p\n",&b);
    printf("%p\n",&c);
    show(a,b,c);
    show(1,2,3);
    f();//2
    f();//3

    //代码块 作用域
    {
        printf("%d\n",a);
        int a=100;
        printf("%d\n",a);
    }
    // 出了代码块,里面的a 变量就销毁了
    int i=100;
    for (int i=0;i<10;i++){
        //循环体里的 局部
        printf("%d\n",i);
    }
    //100
    printf("%d\n",i);

    // register变量也是在当前函数内{}内
    return 0;

}

#include <string.h>
#include <stdlib.h>


// 1. const 修饰的全局变量在常量区,不能需改(直接或间接) 
// 因为内存是只读的 (说明: 所谓只读有些是系统本身设置为只读了
//  有些是我们的编译器设置为只读了)
// 看看 后面的 字符串常量 修改问题

const int a=10;
void test()
{
    // 直接修改
    // a=20;//报错
    // 间接修改
    int *p=(int *)&a;
    *p=20;  //无法修改
    printf("%d\n",a);
}
int main(int argc, char const *argv[])
{

    printf("%p\n",&a); //常量区
    // test();
    puts("==========");
    int c=99;
    printf("%p\n",&c); //栈区

    // 2. const 修饰的局部变量 在栈区
    const int b=10;  // 也是栈区
    printf("%p\n",&b);
    // b=30;// 报错
    printf("%d\n",b);
    int *p=(int *)&b;
    *p=20; // 没有报错 
    printf("%p\n",p);
    printf("%d\n",b); // mac 下 还是10
    printf("%d\n",*p); // mac 下 是20


    puts("======");
    char *p2="hello";
    char *p3="hello";  //p2 和p3 所指向的地址一样的
    // 但是有些编译器 又不一样了,因为他能修改, 如果能修改 就应该不一样,否则改了一个,另外一个也变了
    // 无法修改,,但是如果用其他编译器 可能就可以修改了
    // 就是有些编译器无法修改,有些可以修改 ANSI C中没有规定
    printf("%p\n",p2); // 常量区
    printf("%p\n",&"hello");// 与上面一样的



    return 0;
}

    内存¶

    预处理：宏定义展开、头文件展开、条件编译，这里并不会检查语法 编译：检查语法，将预处理后文件编译生成汇编文件 汇编：将汇编文件生成目标文件(二进制文件) 链接：将目标文件链接为可执行程序

    存放 CPU 执行的机器指令。通常代码区是可共享的 (即另外的执行程序可以调用它) 使其可共享的目的是对于频繁被执行的程序，只需要在内存中有一份代码即可。 代码区通常是只读的，使其只读的原因是防止程序意外地修改了它的指令。另外，代码区还规划了局部变量的相关信息。

    全局初始化数据区/静态数据区 该区包含了在程序中明确被初始化的全局变量、已经初始化的静态变量 (包括全局静态变量和局部静态变量)和常量数据(如字符串常量)。

    未初始化数据区 存入的是全局未初始化变量和未初始化静态变量。 未初始化数据区的数据在程序开始执行之前被内核初始化为 0 或者空(NULL)

    (2和3 也成为全局静态区: 全局区/静态区/常量区 放的地方 字符串常量 和const修饰的全局变量 放在常量区 常量区的数据一旦初始化,就不能修改,只读的内存区 所以 char *p="hello" 后不能修改的 )

    总体来讲说，程序源代码被编译之后主要分成两种段：程序指令和程序数据。 代码段属于程序指令，而数据域段和.bss段属于程序数据。

    -----那为什么把程序的指令和程序数据分开呢？---------

    1. 程序被load到内存中之后，可以将数据和代码分别映射到两个内存区域。 由于数据区域对进程来说是可读可写的，而指令区域对程序来讲说是只读的， 所以分区之后呢，可以将程序指令区域和数据区域分别设置成可读可写或只读。 这样可以防止程序的指令有意或者无意被修改；

    2. 当系统中运行着多个同样的程序的时候，这些程序执行的指令都是一样的， 所以只需要内存中保存一份程序的指令就可以了， 只是每一个程序运行中数据不一样而已，这样可以节省大量的内存。 比如说之前的Windows Internet Explorer 7.0运行起来之后， 它需要占用112 844KB的内存，它的私有部分数据有大概15 944KB， 也就是说有96 900KB空间是共享的，如果程序中运行了几百个这样的进程， 可以想象共享的方法可以节省大量的内存。

    程序在加载到内存前， 代码区和全局区(data和bss)的大小就是固定的， 程序运行期间不能改变。然后，运行可执行程序， 操作系统把物理硬盘程序load(加载)到内存， 除了根据可执行程序的信息分出代码区(text)、数据区(data)和未初始化数据区(bss)之外， 还额外增加了栈区、堆区。

    加载的是可执行文件代码段，所有的可执行代码都加载到代码区， 这块内存是不可以在运行期间修改的。

    加载的是可执行文件BSS段，位置可以分开亦可以紧靠数据段， 存储于数据段的数据(全局未初始化，静态未初始化数据)的生存周期为整个程序运行过程。 全局初始化数据区/静态数据区(data segment) 加载的是可执行文件数据段，存储于数据段(全局初始化，静态初始化数据，文字常量(只读))的数据的生存周期为整个程序运行过程。

    栈是一种先进后出的内存结构，由编译器自动分配释放， 存放函数的参数值、返回值、局部变量等。 在程序运行过程中实时加载和释放，因此， 局部变量的生存周期为申请到释放该段栈空间。 系统自动释放,不需要用户管理

    (唯一能够手动控制的, 内存要记得释放,小心泄露问题) 堆是一个大容器，它的容量要远远大于栈， 但没有栈那样先进后出的顺序。用于动态内存分配。 堆在内存中位于BSS区和栈区之间。一般由程序员分配和释放， 若程序员不释放，程序结束时由操作系统回收。

    一定要记得释放,否则等你程序结束了,可能就是因为内存没释放导致的.

在内存动态存储区中分配nmemb块长度为size字节的连续区域。
calloc自动将分配的内存  置0。<==========
参数：
nmemb：所需内存单元数量
size：每个内存单元的大小（单位：字节）
返回值：
    成功：分配空间的起始地址
失败：NULL
*/

#include <stdlib.h>
void *realloc(void *ptr, size_t size);
功能：
重新分配用malloc或者calloc函数在堆中分配内存空间的大小。
realloc不会自动清理增加的内存，需要手动清理，如果指定的地址后面有连续的空间，那么就会在已有地址基础上增加内存，如果指定的地址后面没有空间，
那么realloc会重新分配新的连续内存，把旧内存的值拷贝到新内存，同时释放旧内存。
参数：
ptr：为之前用malloc或者calloc分配的内存地址，如果此参数等于NULL，那么和realloc与malloc功能一致
size：为重新分配内存的大小, 单位：字节
返回值：
成功：新分配的堆内存地址
失败：NULL
*/
