#include <string.h>
#include <stdlib.h>
int mai1701(int argc, char const *argv[])
{

    /*对于局部性质的数组：
       1. int a[100];这种写法因为没有初始化，所以100元素都是机器垃圾值；
       2. int a[100]={0,2,3};这种写法前3个元素被分别初始化为0、2、3，其余的都置为0；
       3. int a[100]={0};这种写法将100个元素都初始化为0。
    对于全局或修饰为静态性质的数组：
    若初始化则与以上2、3相同；若不初始化则全部元素自动置0。
    */
    // int arr[10];
    // printf("%d\n",arr[9]);// 没有初始化 所以垃圾值
    int arr[10]={0};
    printf("%p\n",&arr[1]);
    // 因为不知道arr里面的元素的类型,所以用void
    void * p=arr;
    printf("%lu\n",sizeof(p));
    puts("===============");
    *(int *)p=22;
    // (int *)p+1 表示加上一个int 大小, 
    // 该数组是int 类型的数组,所以地址上加上一个int大小的话,正好是
    // 第二个 元素的地址了
    *(char *)((int *)p+1)='a';
    // printf("%lu\n",sizeof(long));  long 是8字节的.
    //(long *) p +1=(int *)p+2 
    printf("%p\n",(int *)p+1); // (long *) p +1 试试看,和arr[1]不一样
    for (int i=0;i<10;i++){
        printf("%d\n",arr[i]);
    }
    return 0;
}

    int b=90;

    p1=&b; // 可以修改指针的值

    #include <stdio.h>
#include <string.h>
#include <stdlib.h>

struct cat{
    char a ;
    int b;
    char c;
    int d;
};
void test(){
    struct cat c={'a',10,'b',20};
    printf("%d\n",c.d);
    c.d=100;
    //8
    printf("&c:%lu\n",sizeof(&c));
    // 相差16 默认就是结构体的所占的空间大小
    printf("%p\n",&c);//0x7ffedff65a30
    printf("%p\n",&c+1);//0x7ffedff65a40
    // 所以可以说 指针前面的类型 ,不过就是给他 +1 的操作做了不同的限制而已
    // 只要是个地址  就可以给一个指针
    int * cc=&c; //可以这样
    printf("%p\n",cc);//0x7ffedff65a30
    printf("%p\n",cc+1);//0x7ffedff65a34

    char *dd=&c;
    // 由于对齐,则加12 找到d的地址,然后需要转换为int 指针,
    // * 指针===会根据指针的类型,找该指针类型大小的空间里的数据,
    printf("d的值为:%d\n",*(int *)(dd+12));

    int *p=NULL;//0
    printf("%d\n",p);//0
    printf("%d\n",p+1);//4

    // 记得类型多大的指针,+1就相差多大
    double *p1=NULL;//0
    printf("%d\n",p1);//0
    printf("%d\n",p1+1);//8

    // 指针所占用的大小都是8
    printf("%lu_%lu\n",sizeof(p),sizeof(p1));
};
int main(int argc, char const *argv[])
{

    // 变量名a 只是一块内存的代称,一个符号而已,
    // 看到一个变量 ,脑子里最好想到的就是一块内存
    // 这里就是把100 这个数据放到a这块内存里了
    int a=100;
    // 上面就是一个变量直接赋值

    // 间接赋值
    int *p=&a;
    *p=20;

    test();
    return 0;
}

    printf("%d\n",*(arr+3-1));
